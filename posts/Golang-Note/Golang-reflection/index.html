Golang 反射

2019年10月20日
| Golang
| 阅读

反射

反射是程序在运行时检查其变量和值并找到其类型的能力

首先, 正常时候, 需要查看程序中某个变量的类型和值会这么写:

package main

import (  
	"fmt"
)

func main() {  
	i := 10
	fmt.Printf("%d %T", i, i)
}

这里的变量 i 在编译的时候, 类型就已经是已知的, 所以我们可以打印他的类型

但如果我们想让程序在运行时可以获得变量类型, 例如如下生成 SQL 查询语句的例子

// 生成 SQL 查询语句
package main

import (  
	"fmt"
)

type order struct {  
	ordId int
	customerId int
}

// 接受数据并生成对应的查询语句
func createQuery(o order) string {
	return fmt.Sprintf("insert into order values(%d, %d)", o.ordId, o.customerId)
}

func main() {  
	o := order{
		ordId:      1234,
		customerId: 567,
	}
	fmt.Println(createQuery(o))
}

接着将 createQuery() 函数泛化一下, 使他适用于任何结构

func createQuery(q interface{}) string

当 createQuery() 函数要接受两种不同的数据结构时

type order struct {  
	ordId      int
	customerId int
}

type order struct {  
	name string
	id int
    address string
    salary int
    country string
}

由于 createQuery() 函数需要生成不同长度的查询语句, 需要在语句的 value 填充部分知道传递给它的 struct 的类型, 那么这里需要用到反射的知识

在具体写之前, 需要了解一下 reflect 包中的几种类型和方法

反射类型和反射值

interface{} 的具体类型由 reflect.Type 表示, 基础值由 reflect.Value 表示

对应的两个函数为 reflect.TypeOF() 和 reflect.ValueOf()

还有一种更重要的类型称为 Kind()

// 反射类型和反射值举例
package main

import (
	"fmt"
	"reflect"
)

type order struct {
	ordId      int
	customerId int
}

func createQuery(q interface{}) {
	t := reflect.TypeOf(q)  // 获取具体的实际类型
	v := reflect.ValueOf(q)  // 获取具体的值
    k := t.Kind()  // 获取类型的具体信息
	fmt.Println("Type ", t)
	fmt.Println("Value ", v)
    fmt.Println("Kind ", k)
}

func main() {
	o := order{
		ordId:      456,
		customerId: 56,
	}

	createQuery(o)
}

NumField() 方法返回结构中的字段数, 而 Field(i int) 返回第 i 个字段的 reflect.Value

package main

import (
	"fmt"
	"reflect"
)

type order struct {
	ordId      int
	customerId int
}

func createQuery(q interface{}) {
	if reflect.ValueOf(q).Kind() == reflect.Struct {
		v := reflect.ValueOf(q)
		fmt.Println("Number of fields", v.NumField())

		for i := 0; i < v.NumField(); i++ {
			fmt.Printf("Field:%d type:%T value:%v\n", i, v.Field(i), v.Field(i))
		}
	}

}

func main() {
	o := order{
		ordId:      456,
		customerId: 56,
	}
	
	createQuery(o)
}

Int() 和 String() 方法分别将 reflect.Value 提取为 int64 和 string

package main

import (
	"fmt"
	"reflect"
)

func main() {
	a := 56
	x := reflect.ValueOf(a).Int()
	fmt.Printf("type:%T value:%v\n", x, x)

	b := "Naveen"
	y := reflect.ValueOf(b).String()
	fmt.Printf("type:%T value:%v\n", y, y)
}

结合以上方法, 我们可以实现对 createQuery() 函数的一个泛化:

package main

import (
	"fmt"
	"reflect"
)

type order struct {
	ordId      int
	customerId int
}

type employee struct {
	name    string
	id      int
	address string
	salary  int
	country string
}

func createQuery(q interface{}) {
	if reflect.ValueOf(q).Kind() == reflect.Struct {
		t := reflect.TypeOf(q).Name()  // 获取数据的类型名称 order 和 employee
		query := fmt.Sprintf("insert into %s values(", t)
		v := reflect.ValueOf(q)
		
		for i := 0; i < v.NumField(); i++ {
			switch v.Field(i).Kind() {
			case reflect.Int:
				if i == 0 {
					query = fmt.Sprintf("%s%d", query, v.Field(i).Int())
				} else {
					query = fmt.Sprintf("%s, %d", query, v.Field(i).Int())
				}
			case reflect.String:
				if i == 0 {
					query = fmt.Sprintf("%s\"%s\"", query, v.Field(i).String())
				} else {
					query = fmt.Sprintf("%s, \"%s\"", query, v.Field(i).String())
				}
			default:
				fmt.Println("Unsupported type")
				return
			}
		}

		fmt.Println("query is ", query)

		query = fmt.Sprintf("%s)", query)
		fmt.Println(query)

		return

	}

	fmt.Println("unsupported type")
}

func main() {
	o := order{
		ordId:      456,
		customerId: 56,
	}

	createQuery(o)

	e := employee{
		name:    "Naveen",
		id:      565,
		address: "Coimbatore",
		salary:  90000,
		country: "India",
	}

	createQuery(e)

	i := 90
	createQuery(i)
}
